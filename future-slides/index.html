    public interface CompletionStage<V> {

        CompletionStage<Void>	thenAccept(Consumer<? super T> action);//side effects, println, store to database, send http response to client

        CompletionStage<U> handle(BiFunction<? super T,Throwable,? extends U> fn);//in Play Java known as recover, either value is null or exception is null
        CompletionStage<Void> whenComplete(BiFunction<? super T,Throwable,? extends U> fn);//callback variant

         CompletionStage<U>	thenCompose(Function<? super T,? extends CompletionStage<U>> fn);//if this thing is done, call fn the result of this ant return the result of fn, in scala and play flatMap
         CompletableFuture<V>	thenCombine(CompletionStage<? extends U> other, BiFunction<? super T,? super U,? extends V> fn);//other is another independend future, fn maps the result of this and other into a new result, situation where in scala/play is uses flatmap
    }
